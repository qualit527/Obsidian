TABLE file.ctime AS "创建时间", file.mtime AS "修改时间"
FROM ""
WHERE !contains(file.path, "04_Weekly/") AND !contains(file.path, "00_Inbox/") AND !contains(file.path, "Templates/")
	AND (
	(file.mtime >= date(2025-07-12) AND file.mtime <= date(2025-07-19)) OR
	(file.ctime >= date(2025-07-12) AND file.ctime <= date(2025-07-19))
	)
SORT file.mtime DESC
